    private void startTimer() {
        //Scanner sc = new Scanner(System.in);
        //System.out.print("Input seconds => : ");
        //String secs = sc.nextLine();
        String secs = "5";
        int delay = 1000;
        int period = 1000;
        timer = new Timer();
        interval = Integer.parseInt(secs);
    
        timer.scheduleAtFixedRate(new TimerTask() {

            @Override
            public void run() {
                
                jLabel2.setText("Kalan Zaman: "+String.valueOf(setInterval())+" saniye");
                if(interval == 0) {
                    getProcesses();
                }

            }
        }, delay, period);
        //getProcesses();
    }
    private static int setInterval() {
        if (interval == 1)
            timer.cancel();           
        return --interval;
    }

    public void getProcesses() {
        try {
            String line;
            //Process p = Runtime.getRuntime().exec("ps -e");
            Process p = Runtime.getRuntime().exec
            (System.getenv("windir") +"\\system32\\"+"tasklist.exe");

            BufferedReader input;
                input = new BufferedReader(new InputStreamReader(p.getInputStream()));
            
                while ((line = input.readLine()) != null) {            
                    System.out.println(line); //<-- Parse data here.           
                    if(line.toLowerCase().contains(processNameToCompare)) {
                        JOptionPane.showMessageDialog(null, "Tebrikler! Bir sonraki asamaya geçebilirsiniz");                  
                        jButton5.setEnabled(true);
                        jButton5.setIcon(nextEnabled);
                        jButton6.setEnabled(false);
                        jButton7.setEnabled(false);
                        getTasksNameFromXML(2,taskPosition);
                        taskPosition++;     
                        flag = true;                       
                    }                        
                }
                if(flag  == false) {
                    JOptionPane.showMessageDialog(null, "Üzgünüm, basarisiz oldunuz.Tekrar deneyin!");
                } else {
                    if(taskPosition-1 == nodeLstProcess.getLength()) {
                    JOptionPane.showMessageDialog(null, "Bütün asamalari tamamladiniz!");
                    jButton6.setEnabled(false);

    // Task isimlerini alir. usage = 1 ise sinifseviyesine göre task ismi alir, usage = 2 ise processId'ye bakar 
    // ve processId id'ye esitse jLabel1 ve jLabel3'e ilgili degerleri atar deger atar.
    private void getTasksNameFromXML(int usage, int processId) {
        try {           
           dbf = DocumentBuilderFactory.newInstance();
           db = dbf.newDocumentBuilder();
           doc = db.parse(new File("tasks.xml"));
           doc.normalize();

           nodeLstProcess = doc.getElementsByTagName("item");          
           
           for (int i = 0; i < nodeLstProcess.getLength(); i++) {             
               
               Node rootNode = nodeLstProcess.item(i);
               Element myElmnt = (Element) rootNode;

               //Get id of current item
               NodeList idElmntLst = myElmnt.getElementsByTagName("id");
               Element idElmnt = (Element) idElmntLst.item(0);
               NodeList idN = idElmnt.getChildNodes();
               id = ((Node) idN.item(0)).getNodeValue();
               
               //Get name of current item
               NodeList nameElmntLst = myElmnt.getElementsByTagName("name");
               Element nameElmnt = (Element) nameElmntLst.item(0);
               NodeList nameN = nameElmnt.getChildNodes();
               name = ((Node) nameN.item(0)).getNodeValue();

               //Get hints of current item
               NodeList hintsElmntLst = myElmnt.getElementsByTagName("hints");
               Element hintsElmnt = (Element) hintsElmntLst.item(0);
               NodeList hintsN = hintsElmnt.getChildNodes();
               hints = ((Node) hintsN.item(0)).getNodeValue();

               //Get relatedTasks of current item
               NodeList relatedTasksElmntLst = myElmnt.getElementsByTagName("relatedTasks");
               Element relatedTasksElmnt = (Element) relatedTasksElmntLst.item(0);
               NodeList relatedTasksN = relatedTasksElmnt.getChildNodes();
               relatedTasks = ((Node) relatedTasksN.item(0)).getNodeValue();

               //Get processName of current item
               NodeList processNameElmntLst = myElmnt.getElementsByTagName("processName");
               Element processNameElmnt = (Element) processNameElmntLst.item(0);
               NodeList processNameN = processNameElmnt.getChildNodes();
               processName = ((Node) processNameN.item(0)).getNodeValue();
               
               if(usage == 1) {         
                   if (id.equals(sinifSeviyesiXML)) {         
                    taskName = name;
                   }    
               } else if(usage == 2) {
                   if (id.equals(String.valueOf(processId))) {         
                       jLabel1.setText("GÖREV "+id+":");
                       jLabel3.setText(name);
                       processNameToCompare = processName;
                   }
               }
               
           }  
        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "Görev listesi dosya hatasi!");
        }
}
